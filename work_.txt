// соберем кнтейнер с mysql v8
./Dockerfile
FROM mysql:8.0
# # use MySQL v8

RUN mkdir -p /slaveMysql/script/
# # script dir`s

RUN mkdir -p /slaveMysql/data/
# # database storage

WORKDIR /slaveMysql/script/
#

COPY ./my.cnf /slaveMysql/script/
#

# setup ENV param
ENV TZ Europe/Moscow

# декларируем сетевой порт
EXPOSE 3406

# CMD ["mysql", ""]
# # run in system shell
# CMD ["mysql","/путь/к_скрипту/с_импортом"]

# # run without shell
# ENTRYPOINT ["", ""]


// https://dev.mysql.com/doc/refman/8.0/en/docker-mysql-getting-started.html
// docker run --name=mysql80 \
//   --mount type=bind,src=/path-on-host-machine/my.cnf,dst=/etc/my.cnf \

//   -d mysql/mysql-server:8.0


// соберем в соответствии с Dockerfile контейнер -t <image name>
docker build -t mysqlSlave .

// запуск контейнера -d в фоне иначе в терминал пишутся логи
// --rm удалить контейнер после остановки
// -p <порт на host>:<порт в контейнере> 38:33
// -e TZ=Europe/Moscow setup переменных окружения
docker run --name myTest -p 3406:3406 -d --rm mysqlSlave

// остановить контейнер
docker stop <container id>|<name>

// удалить все контейнеры
// // docker rm $(docker ps -qa)
// удалить отработавший контейнер
docker rm <name>|<container id>

// выведет ппострочно все <container id>
// // docker ps -a -q
// список контейнеров работающих, с ключем -a все(в том числе и остановленных)
docker ps

// подключение внешнего хранилища (контейнер Read Only)
// т.е. запущеные приложения не могут изменять внутр. содержимое
// монтировать можно двумя способами
// -v <путь к месту хранения на Host>:<path in docker image>
docker run --name myTest -p 3406:3406 -v <путь к месту хранения на Host>:<path in docker image> -d --rm mysqlSlave
// или создаем точку монтирования
//  docker volume ls -- покажет существующие
//  docker volume create <имя> -- создаст пустой волюм который
// при запуске контейнера будет сопоставлен WORKDIR
docker run --name myTest -p 3406:3406 -v <имя волюма>:<path in docker image> -d --rm mysqlSlave
